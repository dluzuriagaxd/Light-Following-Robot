  <script>
    // Configuración de las posiciones por defecto (para alineación perfecta/ensamblado)
    const originalPositions = {
      'bloque-on-start':          { x: 50, y: 50 },
      'bloque-forever-contenedor': { x: 50, y: 230 },
      'bloque-led-high':          { x: 66, y: 298 },
      'bloque-wait-1':            { x: 66, y: 344 },
      'bloque-led-low':           { x: 66, y: 390 },
      'bloque-wait-2':            { x: 66, y: 436 }
    };

    // Configuración de posiciones dispersas ("Exploded" para verlos separados)
    const explodedPositions = {
      'bloque-on-start':          { x: 50, y: 80 },
      'bloque-forever-contenedor': { x: 340, y: 80 },
      'bloque-led-high':          { x: 640, y: 50 },
      'bloque-wait-1':            { x: 640, y: 130 },
      'bloque-led-low':           { x: 640, y: 210 },
      'bloque-wait-2':            { x: 640, y: 290 }
    };

    // Inicializar posiciones de los elementos SVG
    function initPositions(positions) {
      Object.keys(positions).forEach(id => {
        const el = document.getElementById(id);
        if (el) {
          // Asignar traducción de posición inicial
          el.setAttribute('transform', `translate(${positions[id].x}, ${positions[id].y})`);
        }
      });
    }

    // Inicializar la interfaz en modo ensamblado por defecto
    window.onload = function() {
      initPositions(originalPositions);
      setupDragAndDrop();
    };

    // Implementación del Drag & Drop nativo y sensible a touch
    function setupDragAndDrop() {
      const svg = document.getElementById('workspace-svg');
      let selectedElement = null;
      let offset = { x: 0, y: 0 };

      // Capturar eventos de mouse y touch
      svg.addEventListener('mousedown', startDrag);
      svg.addEventListener('mousemove', drag);
      svg.addEventListener('mouseup', endDrag);
      svg.addEventListener('mouseleave', endDrag);

      svg.addEventListener('touchstart', startDrag, { passive: false });
      svg.addEventListener('touchmove', drag, { passive: false });
      svg.addEventListener('touchend', endDrag);

      // Convertir coordenadas de pantalla a coordenadas internas del SVG
      function getMousePosition(evt) {
        const CTM = svg.getScreenCTM();
        if (evt.touches && evt.touches.length > 0) {
          evt = evt.touches[0];
        }
        return {
          x: (evt.clientX - CTM.e) / CTM.a,
          y: (evt.clientY - CTM.f) / CTM.d
        };
      }

      function startDrag(evt) {
        let target = evt.target;
        // Buscar el ancestro que sea un grupo draggable
        while (target && target !== svg) {
          if (target.classList.contains('draggable')) {
            selectedElement = target;
            break;
          }
          target = target.parentNode;
        }

        if (selectedElement) {
          evt.preventDefault();
          const coord = getMousePosition(evt);
          
          // Asegurar que exista una matriz de transformación
          let transform = selectedElement.transform.baseVal;
          if (transform.numberOfItems === 0 || transform.getItem(0).type !== SVGTransform.SVG_TRANSFORM_TRANSLATE) {
            let translate = svg.createSVGTransform();
            translate.setTranslate(0, 0);
            selectedElement.transform.baseVal.insertItemBefore(translate, 0);
          }
          
          const trans = transform.getItem(0);
          offset.x = coord.x - trans.matrix.e;
          offset.y = coord.y - trans.matrix.f;

          // Traer elemento seleccionado al frente visualmente
          selectedElement.parentNode.appendChild(selectedElement);
        }
      }

      // Desplazamiento del elemento
      function drag(evt) {
        if (selectedElement) {
          evt.preventDefault();
          const coord = getMousePosition(evt);
          const trans = selectedElement.transform.baseVal.getItem(0);
          trans.setTranslate(coord.x - offset.x, coord.y - offset.y);
        }
      }

      function endDrag(evt) {
        selectedElement = null;
      }
    }

    // Funciones del Panel de Control
    const notification = document.getElementById('notification');
    
    function showNotification(msg) {
      notification.innerText = msg;
      notification.classList.remove('opacity-0', 'translate-y-[-10px]', 'pointer-events-none');
      notification.classList.add('opacity-100', 'translate-y-0');
      
      setTimeout(() => {
        notification.classList.remove('opacity-100', 'translate-y-0');
        notification.classList.add('opacity-0', 'translate-y-[-10px]', 'pointer-events-none');
      }, 2500);
    }

    // Botón Ensamblar
    document.getElementById('btn-assemble').addEventListener('click', () => {
      initPositions(originalPositions);
      showNotification('Bloques acoplados a la perfección.');
    });

    // Botón Separar
    document.getElementById('btn-explode').addEventListener('click', () => {
      initPositions(explodedPositions);
      showNotification('Bloques independizados listos para mover.');
    });

    // Botón Cambiar Tema
    const canvasWrapper = document.getElementById('canvas-wrapper');
    const themeIcon = document.getElementById('theme-icon');
    const btnTheme = document.getElementById('btn-theme');
    let isDark = false;

    btnTheme.addEventListener('click', () => {
      isDark = !isDark;
      if (isDark) {
        canvasWrapper.className = 'w-full h-full canvas-grid-dark transition-all duration-300';
        btnTheme.innerHTML = '☀️ Ligero';
        btnTheme.className = 'px-3 py-2 bg-slate-800 hover:bg-slate-700 text-slate-100 rounded-lg text-sm font-medium transition flex items-center gap-2';
      } else {
        canvasWrapper.className = 'w-full h-full canvas-grid-light transition-all duration-300';
        btnTheme.innerHTML = '🌙 Oscuro';
        btnTheme.className = 'px-3 py-2 bg-slate-100 hover:bg-slate-200 text-slate-700 rounded-lg text-sm font-medium transition flex items-center gap-2';
      }
    });

    // Botón Descargar SVG (Exportación limpia del estado actual del lienzo)
    document.getElementById('btn-download').addEventListener('click', () => {
      const svgEl = document.getElementById('workspace-svg');
      
      // Duplicamos el nodo SVG para limpiarlo antes de descargar
      const cleanSvg = svgEl.cloneNode(true);
      
      // Removemos los atributos de pantalla interactiva, definiendo dimensiones fijas idóneas para exportación
      cleanSvg.setAttribute('width', '1000');
      cleanSvg.setAttribute('height', '700');
      
      const svgString = new XMLSerializer().serializeToString(cleanSvg);
      const blob = new Blob([svgString], { type: 'image/svg+xml;charset=utf-8' });
      const URL = window.URL || window.webkitURL || window;
      const blobURL = URL.createObjectURL(blob);
      
      const a = document.createElement('a');
      a.href = blobURL;
      a.download = 'bloques_programacion_corregidos.svg';
      document.body.appendChild(a);
      a.click();
      
      document.body.removeChild(a);
      URL.revokeObjectURL(blobURL);
      
      showNotification('Archivo SVG descargado con éxito.');
    });
  </script>
